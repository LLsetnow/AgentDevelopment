# 工具开发指南

## 概述
工具（Tool）是 Agent 调用外部函数的方式，用于 Function Calling 功能。

## 工具模板

```python
# tools/my_tool.py
from typing import Dict, Any, Optional, List
import json

def tool_wrapper(
    name: str,
    description: str,
    parameters: Dict[str, Any],
    function: callable
) -> Dict[str, Any]:
    """
    工具包装器，将函数包装为标准格式
    
    Args:
        name: 工具名称
        description: 工具描述
        parameters: 参数定义（JSON Schema 格式）
        function: 工具函数
        
    Returns:
        工具定义字典
    """
    return {
        "type": "function",
        "function": {
            "name": name,
            "description": description,
            "parameters": {
                "type": "object",
                "properties": parameters,
                "required": list(parameters.keys())
            }
        },
        "handler": function
    }

# ==================== 示例工具 ====================

def search_web(query: str, num_results: int = 5) -> str:
    """
    网络搜索工具
    
    Args:
        query: 搜索关键词
        num_results: 返回结果数量
        
    Returns:
        搜索结果 JSON 字符串
    """
    # 实现搜索逻辑
    results = [
        {"title": "示例结果1", "url": "https://example.com/1"},
        {"title": "示例结果2", "url": "https://example.com/2"}
    ]
    return json.dumps(results, ensure_ascii=False)

def calculate(expression: str) -> str:
    """
    计算工具
    
    Args:
        expression: 数学表达式（如 "2 + 3 * 4"）
        
    Returns:
        计算结果
    """
    try:
        result = eval(expression)
        return str(result)
    except Exception as e:
        return f"计算错误: {str(e)}"

def read_file(filepath: str, encoding: str = "utf-8") -> str:
    """
    读取文件工具
    
    Args:
        filepath: 文件路径
        encoding: 文件编码
        
    Returns:
        文件内容
    """
    try:
        with open(filepath, 'r', encoding=encoding) as f:
            return f.read()
    except Exception as e:
        return f"读取文件失败: {str(e)}"

def write_file(filepath: str, content: str, mode: str = "w") -> str:
    """
    写入文件工具
    
    Args:
        filepath: 文件路径
        content: 文件内容
        mode: 写入模式（'w' 覆盖, 'a' 追加）
        
    Returns:
        操作结果
    """
    try:
        with open(filepath, mode, encoding='utf-8') as f:
            f.write(content)
        return f"文件写入成功: {filepath}"
    except Exception as e:
        return f"写入文件失败: {str(e)}"

def get_current_time(format: str = "%Y-%m-%d %H:%M:%S") -> str:
    """
    获取当前时间工具
    
    Args:
        format: 时间格式
        
    Returns:
        格式化的时间字符串
    """
    from datetime import datetime
    return datetime.now().strftime(format)

# ==================== 工具注册 ====================

def get_available_tools() -> List[Dict[str, Any]]:
    """
    获取所有可用工具
    
    Returns:
        工具定义列表
    """
    return [
        tool_wrapper(
            name="search_web",
            description="在网络上搜索信息",
            parameters={
                "query": {
                    "type": "string",
                    "description": "搜索关键词"
                },
                "num_results": {
                    "type": "integer",
                    "description": "返回结果数量，默认5"
                }
            },
            function=search_web
        ),
        tool_wrapper(
            name="calculate",
            description="计算数学表达式",
            parameters={
                "expression": {
                    "type": "string",
                    "description": "数学表达式，如 '2 + 3 * 4'"
                }
            },
            function=calculate
        ),
        tool_wrapper(
            name="read_file",
            description="读取文本文件内容",
            parameters={
                "filepath": {
                    "type": "string",
                    "description": "文件路径"
                },
                "encoding": {
                    "type": "string",
                    "description": "文件编码，默认utf-8"
                }
            },
            function=read_file
        ),
        tool_wrapper(
            name="write_file",
            description="将内容写入文件",
            parameters={
                "filepath": {
                    "type": "string",
                    "description": "文件路径"
                },
                "content": {
                    "type": "string",
                    "description": "文件内容"
                },
                "mode": {
                    "type": "string",
                    "description": "写入模式：'w' 覆盖，'a' 追加，默认'w'"
                }
            },
            function=write_file
        ),
        tool_wrapper(
            name="get_current_time",
            description="获取当前时间",
            parameters={
                "format": {
                    "type": "string",
                    "description": "时间格式，如 '%Y-%m-%d %H:%M:%S'"
                }
            },
            function=get_current_time
        )
    ]

# ==================== 工具调用器 ====================

class ToolExecutor:
    """工具执行器"""
    
    def __init__(self):
        self.tools = {tool["function"]["name"]: tool for tool in get_available_tools()}
    
    def execute(self, tool_name: str, arguments: Dict[str, Any]) -> str:
        """
        执行工具
        
        Args:
            tool_name: 工具名称
            arguments: 参数字典
            
        Returns:
            执行结果
        """
        if tool_name not in self.tools:
            return f"错误: 工具 '{tool_name}' 不存在"
        
        tool = self.tools[tool_name]
        handler = tool["handler"]
        
        try:
            result = handler(**arguments)
            return result
        except Exception as e:
            return f"执行失败: {str(e)}"
    
    def list_tools(self) -> List[str]:
        """列出所有工具"""
        return list(self.tools.keys())
    
    def get_tool_info(self, tool_name: str) -> Optional[Dict[str, Any]]:
        """获取工具信息"""
        if tool_name in self.tools:
            return self.tools[tool_name]
        return None
```

## 在 Agent 中集成工具

```python
# simple.py 中添加工具支持
from tools.my_tool import ToolExecutor

class DeepSeekAgent:
    def __init__(self, api_key: str):
        # ... 现有代码 ...
        self.tool_executor = ToolExecutor()
    
    def _build_messages(self) -> List[dict]:
        """构建消息列表，包含工具定义"""
        messages = []
        if self.system_prompt:
            messages.append({"role": "system", "content": self.system_prompt})
        messages.extend([{"role": msg.role, "content": msg.content} for msg in self.messages])
        
        # 添加工具定义
        tools_def = [
            tool["function"] for tool in self.tool_executor.tools.values()
        ]
        return messages, tools_def
    
    def chat_with_tools(self, user_message: str) -> str:
        """
        带工具调用的对话
        
        Args:
            user_message: 用户消息
            
        Returns:
            Agent 的回复
        """
        messages, tools = self._build_messages()
        
        headers = {
            "Content-Type": "application/json",
            "Authorization": f"Bearer {self.api_key}"
        }
        
        payload = {
            "model": self.model,
            "messages": messages + [{"role": "user", "content": user_message}],
            "tools": tools if tools else None,
            "temperature": 0.7
        }
        
        response = requests.post(self.api_url, headers=headers, json=payload)
        result = response.json()
        
        # 检查是否需要调用工具
        if "tool_calls" in result["choices"][0]["message"]:
            tool_calls = result["choices"][0]["message"]["tool_calls"]
            
            # 执行所有工具调用
            for tool_call in tool_calls:
                tool_name = tool_call["function"]["name"]
                arguments = json.loads(tool_call["function"]["arguments"])
                tool_result = self.tool_executor.execute(tool_name, arguments)
                
                # 将工具结果添加到对话中
                messages.append({
                    "role": "tool",
                    "tool_call_id": tool_call["id"],
                    "content": tool_result
                })
            
            # 再次调用 API 获取最终回复
            response = requests.post(self.api_url, headers=headers, json=payload)
            result = response.json()
        
        return result["choices"][0]["message"]["content"]
```

## 使用示例

```python
# 创建 Agent
agent = DeepSeekAgent(api_key="your_api_key")

# 使用带工具的对话
response = agent.chat_with_tools("帮我计算 2 * 3 + 4 的结果")
print(response)

response = agent.chat_with_tools("读取 config.yaml 文件的内容")
print(response)
```

## 工具开发最佳实践

1. **安全第一**: 谨慎处理文件操作和系统命令
2. **输入验证**: 验证所有输入参数
3. **错误处理**: 捕获异常并返回友好信息
4. **返回格式**: 统一返回字符串格式
5. **参数文档**: 详细描述每个参数
6. **幂等性**: 尽量保证工具的幂等性
