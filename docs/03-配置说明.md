## 配置文件加载

```python
import yaml
from typing import Dict, Any
from pathlib import Path

class Config:
    """配置管理器"""
    
    def __init__(self, config_path: str = "config.yaml"):
        self.config_path = config_path
        self.config = self._load_config()
    
    def _load_config(self) -> Dict[str, Any]:
        """加载配置文件"""
        try:
            with open(self.config_path, 'r', encoding='utf-8') as f:
                return yaml.safe_load(f) or {}
        except FileNotFoundError:
            print(f"警告: 配置文件 {self.config_path} 不存在，使用默认配置")
            return self._get_default_config()
        except Exception as e:
            print(f"加载配置文件失败: {str(e)}")
            return self._get_default_config()
    
    def _get_default_config(self) -> Dict[str, Any]:
        """获取默认配置"""
        return {
            "api": {
                "key": "",
                "model": "deepseek-chat",
                "base_url": "https://api.deepseek.com"
            },
            "request": {
                "temperature": 0.7,
                "max_tokens": 2000
            },
            "timeout": {
                "api_timeout": 60
            },
            "logging": {
                "level": "INFO",
                "file": "logs/agent.log"
            }
        }
    
    def get(self, key: str, default: Any = None) -> Any:
        """
        获取配置值
        
        Args:
            key: 配置键，支持点号分隔（如 "api.key"）
            default: 默认值
            
        Returns:
            配置值
        """
        keys = key.split('.')
        value = self.config
        
        for k in keys:
            if isinstance(value, dict) and k in value:
                value = value[k]
            else:
                return default
        
        return value
    
    def set(self, key: str, value: Any):
        """
        设置配置值
        
        Args:
            key: 配置键
            value: 配置值
        """
        keys = key.split('.')
        config = self.config
        
        for k in keys[:-1]:
            if k not in config:
                config[k] = {}
            config = config[k]
        
        config[keys[-1]] = value
    
    def save(self, path: str = None):
        """
        保存配置到文件
        
        Args:
            path: 保存路径，默认为原配置文件路径
        """
        save_path = path or self.config_path
        with open(save_path, 'w', encoding='utf-8') as f:
            yaml.dump(self.config, f, allow_unicode=True, default_flow_style=False)
```

## 使用配置

```python
# 初始化配置
config = Config()

# 获取配置
api_key = config.get("api.key")
model = config.get("api.model")
temperature = config.get("request.temperature")

# 设置配置
config.set("request.temperature", 0.8)

# 保存配置
config.save()

# 使用配置初始化 Agent
agent = DeepSeekAgent(
    api_key=config.get("api.key"),
    model=config.get("api.model")
)
agent.temperature = config.get("request.temperature")
```

## 环境变量支持

也可以使用环境变量来覆盖配置：

```python
import os

class Config:
    def __init__(self, config_path: str = "config.yaml"):
        self.config_path = config_path
        self.config = self._load_config()
        self._load_from_env()
    
    def _load_from_env(self):
        """从环境变量加载配置"""
        # API Key 优先从环境变量读取
        if "DEEPSEEK_API_KEY" in os.environ:
            self.config["api"]["key"] = os.environ["DEEPSEEK_API_KEY"]
        
        # 其他配置
        if "DEEPSEEK_MODEL" in os.environ:
            self.config["api"]["model"] = os.environ["DEEPSEEK_MODEL"]
        
        if "DEEPSEEK_TEMPERATURE" in os.environ:
            self.config["request"]["temperature"] = float(os.environ["DEEPSEEK_TEMPERATURE"])
```

## 安全建议

1. **不要提交 config.yaml**: 使用 `.gitignore` 忽略
2. **使用环境变量**: 敏感信息通过环境变量传递
3. **权限控制**: 设置配置文件适当的读写权限
4. **定期轮换**: 定期更换 API Key
5. **最小权限**: 使用最小权限原则配置
