## 4. 性能优化

### 缓存机制
```python
from functools import lru_cache

class CacheManager:
    def __init__(self):
        self.cache = {}
        self.expiration = 3600  # 1小时
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        if key in self.cache:
            data, timestamp = self.cache[key]
            if time.time() - timestamp < self.expiration:
                return data
        return None
    
    def set(self, key: str, value: Any):
        """设置缓存"""
        self.cache[key] = (value, time.time())

# 使用装饰器缓存
@lru_cache(maxsize=100)
def expensive_function(x: int) -> int:
    return x * x
```

### 异步处理
```python
import asyncio

async def async_call_api(url: str, payload: dict) -> dict:
    """异步 API 调用"""
    async with aiohttp.ClientSession() as session:
        async with session.post(url, json=payload) as response:
            return await response.json()

async def process_tasks(tasks: list):
    """批量处理任务"""
    results = await asyncio.gather(*tasks)
    return results
```

```python
def validate_input(user_input: str) -> bool:
    """验证用户输入"""
    if not user_input:
        raise ValueError("输入不能为空")
    
    if len(user_input) > 10000:
        raise ValueError("输入过长")
    
    # 检查危险字符
    dangerous_chars = ['<script', 'javascript:', 'eval(']
    for char in dangerous_chars:
        if char in user_input.lower():
            raise ValueError("输入包含危险字符")
    
    return True
```

### 权限控制
```python
# 文件操作前检查
def safe_write_file(filepath: str, content: str):
    """安全写入文件"""
    # 检查文件路径
    allowed_dirs = ["/data", "/tmp"]
    if not any(filepath.startswith(d) for d in allowed_dirs):
        raise PermissionError("不允许的文件路径")
    
    # 检查文件大小
    if len(content) > 10 * 1024 * 1024:  # 10MB
        raise ValueError("文件过大")
    
    # 写入文件
    with open(filepath, 'w') as f:
        f.write(content)
```

### 性能监控
```python
import time

def monitor_performance(func):
    """性能监控装饰器"""
    def wrapper(*args, **kwargs):
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        
        duration = end_time - start_time
        logger.info(f"{func.__name__} 执行耗时: {duration:.2f}秒")
        
        if duration > 10:
            logger.warning(f"{func.__name__} 执行时间过长")
        
        return result
    return wrapper

@monitor_performance
def expensive_operation():
    # 耗时操作
    pass
```
